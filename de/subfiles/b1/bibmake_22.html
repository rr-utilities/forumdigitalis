2.2 Wichtige Eintragsarten

In der .bib-Datei, in der die Quellen eingetragen werden, werden sie in einzelnen Eintragstypen unterteilt. Jeder dieser Typen haben einzelne unterschiedliche Komponente.

@book{Quelle1, % Buch (Monografie) author = {}, title = {}, publisher = {}, year = {}, address = {}, edition = {}, isbn = {} }
@article{Quelle2, % Zeitschriftenartikel author = {}, title = {}, journal = {}, year = {}, volume = {}, number = {}, pages = {}, doi = {} }
@inproceedings{Quelle3, % Einzelner Beitrag in Konferenzband author = {}, title = {}, booktitle = {}, year = {}, editor = {}, pages = {}, publisher = {}, address = {} }
@incollection{Quelle4, % Kapitel in Sammelband author = {}, title = {}, booktitle = {}, editor = {}, publisher = {}, address = {}, year = {}, pages = {} }
@proceedings{Quelle5, % Gesamter Konferenzband editor = {}, title = {}, year = {}, publisher = {}, address = {} }
@report{Quelle6, % Bericht (intern, öffentlich, technisch) author = {}, title = {}, institution = {}, year = {}, number = {}, address = {}, url = {} }
@techreport{Quelle7, % Technischer Bericht (spezieller Report-Typ) author = {}, title = {}, institution = {}, year = {}, number = {}, address = {} }
@phdthesis{Quelle8, % Dissertation author = {}, title = {}, school = {}, year = {}, address = {} }
@mastersthesis{Quelle9, % Masterarbeit author = {}, title = {}, school = {}, year = {}, address = {} }
@misc{Quelle10, % Alles, was keiner Kategorie klar entspricht author = {}, title = {}, year = {}, url = {}, note = {} }
@online{Quelle11, % Nur bei biblatex (nicht bei BibTeX) author = {}, title = {}, year = {}, url = {}, urldate = {} }
@manual{Quelle12, % Bedienungsanleitungen, Handbücher title = {}, author = {}, organization = {}, year = {}, url = {} }
@unpublished{Quelle13, % Unveröffentlichte Dokumente author = {}, title = {}, note = {}, year = {} }
